\subsection{Tạo chương trình học}

    \centering

    \caption{Sơ đồ use-case cho chức năng "Tạo lớp học"}
    \label{fig:create_class_activity}
\end{figure}
Activity diagram - Tạo lớp học mô tả quy trình \textbf{tạo lớp học} giữa hai tác nhân: \textbf{Người dùng} và \textbf{Hệ thống}. 
Luồng hoạt động diễn ra như sau:

\begin{itemize}
    \item Người dùng bắt đầu quy trình và nhấn nút \textit{“Tạo lớp”}.
    \item Hệ thống hiển thị form nhập thông tin lớp học.
    \item Người dùng nhập các thông tin cần thiết cho lớp học.
    \item Sau khi hoàn tất, người dùng nhấn nút \textit{“Lưu”}.
    \item Hệ thống tiếp nhận dữ liệu và tiến hành kiểm tra thông tin.
    \item Tại bước kiểm tra, hệ thống thực hiện hai nhánh xử lý:
    \begin{itemize}
        \item \textbf{Trường hợp thông tin hợp lệ}:
        \begin{itemize}
            \item Hệ thống lưu thông tin lớp học vào cơ sở dữ liệu.
            \item Hệ thống thêm lớp học vào lịch dạy tương ứng.
            \item Hệ thống đồng thời thêm lớp học này vào danh sách lớp để sinh viên có thể đăng ký.
        \end{itemize}
        \item \textbf{Trường hợp thông tin không hợp lệ}:
        \begin{itemize}
            \item Hệ thống hiển thị thông báo lỗi và yêu cầu người dùng nhập lại thông tin.
        \end{itemize}
    \end{itemize}
    \item Quy trình kết thúc khi thông tin lớp được lưu thành công hoặc người dùng sửa lại thông tin theo yêu cầu hệ thống.
\end{itemize}

    \centering

    \caption{Sơ đồ use-case cho chức năng "Quản lí lớp học"}
    \label{fig:manage_class_activity}
\end{figure}
Activity diagram - Quản lý lớp và lịch dạy mô tả quy trình người dùng (tutor) tương tác với hệ thống để quản lý
các lớp học và lịch dạy. Quy trình được mô tả như sau:

\begin{itemize}
    \item Người dùng bắt đầu bằng cách chọn chức năng \textit{“Quản lý lớp và lịch dạy”}.
    \item Hệ thống hiển thị danh sách tất cả các lớp học mà tutor đã tạo trước đó.
    \item Người dùng lựa chọn xem chi tiết một lớp học cụ thể.
    \begin{itemize}
        \item Nếu người dùng không chọn xem chi tiết, quy trình kết thúc.
        \item Nếu người dùng chọn xem chi tiết, hệ thống sẽ hiển thị đầy đủ thông tin của lớp học đó.
    \end{itemize}

    \item Tại giao diện chi tiết lớp học, người dùng có thể:
    \begin{itemize}
        \item Chọn trở về danh sách lớp.
        \item Hoặc chọn chức năng \textit{hủy lớp} hoặc \textit{chỉnh sửa lớp}.
    \end{itemize}

    \item Nếu người dùng chọn chỉnh sửa, hệ thống hiển thị trạng thái lớp để cho phép thay đổi thông tin.
    \item Người dùng tiến hành nhập và sửa đổi các thông tin cần thiết, sau đó chọn \textit{“Lưu”}.
    \item Hệ thống kiểm tra tính hợp lệ của thời gian:
    \begin{itemize}
        \item Nếu thời gian không hợp lệ hoặc trùng lịch, hệ thống thông báo lỗi và yêu cầu người dùng nhập lại.
        \item Nếu thời gian hợp lệ, hệ thống lưu thông tin mới vào cơ sở dữ liệu.
    \end{itemize}

    \item Quy trình kết thúc sau khi thông tin lớp được cập nhật hoặc người dùng thoát khỏi chức năng.
\end{itemize}

Activity diagram - Thiết lập lịch rảnh mô tả quy trình người dùng thiết lập lịch rảnh thông qua hệ thống. 
Quy trình bao gồm hai tác nhân chính: \textbf{Người dùng} và \textbf{Hệ thống}. 
Luồng hoạt động diễn ra như sau:
\begin{itemize}
    \item Người dùng bắt đầu quy trình và chọn chức năng \textit{thiết lập lịch rảnh}.
    \item Hệ thống hiển thị danh sách các lịch rảnh đã có sẵn.
    \item Người dùng chọn thao tác \textit{thêm lịch rảnh}.
    \item Hệ thống hiển thị form thiết lập lịch rảnh mới.
    \item Người dùng chọn thời gian và loại hình phù hợp.
    \item Người dùng nhấn nút \textit{Lưu}.
    \item Hệ thống thực hiện kiểm tra thông tin:
    \begin{itemize}
        \item Nếu không trùng lịch đã tồn tại:
        \begin{itemize}
            \item Hệ thống lưu thông tin vào cơ sở dữ liệu.
            \item Hệ thống gửi thông báo lưu thành công.
        \end{itemize}
        \item Nếu trùng lịch rảnh đã có:
        \begin{itemize}
            \item Hệ thống hiển thị thông báo trùng lịch, yêu cầu người dùng điều chỉnh.
        \end{itemize}
    \end{itemize}
    \item Quy trình kết thúc sau khi thông tin được lưu thành công hoặc sau khi hệ thống 
    hiển thị thông báo trùng lịch.
\end{itemize}
