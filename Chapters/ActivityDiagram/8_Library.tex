\subsection*{Mô tả sơ đồ hoạt động: Tải tài liệu lên}
\begin{itemize}
    \item \textbf{Người dùng} chọn biểu tượng "Tải xuống" trên giao diện ứng dụng.
    \item Sau đó nhập những thông tin cần thiết cho tệp sẽ tải lên.
    \item \textbf{Hệ thống} kiểm tra kích thước và định dạng tệp tin để quyết định xem có cho người dùng tải lên hay không.
    \item Nếu hợp lệ thì chuẩn bị tải tệp lên \textbf{hệ thống}. Trong quá trình tải sẽ phát sinh sự cố nên hệ thống sẽ báo lỗi tới người dùng khi xảy ra.
    \item Nếu \textbf{hệ thống} tải lên thành công sẽ báo về phái người dùng.
\end{itemize}

\subsection*{Mô tả sơ đồ hoạt động: Việc tìm kiếm và xem trước tài liệu}
\begin{itemize}
    \item \textbf{Người dùng} nhập từ khóa trên thanh "Tìm kiếm" và có thể chọn bộ lọc.
    \item \textbf{Hệ thống} sau đó thực hiện truy vấn dựa trên yêu cầu từ \textbf{người dùng}, nếu tìm thấy sẽ trả về danh sách các thông tin của tài liệu 
về phía \textbf{người dùng} và đợi \textbf{người dùng} bấm chọn.
    \item Khi người dùng chọn một thông tin tài liệu, \textbf{hệ thống} thực hiện các kiểm tra cần thiết và truy xuất bộ nhớ để lấy dữ liệu xem trước.
    \item \textbf{Hệ thống} sasu đó trả về kết quả và hiển thị giao diện xem trước tài liệu cho \textbf{người dùng}.
    \item Giao diện xem trước có thêm nút tải xuống nếu tài liệu cho phép tải về.
    \item Trong quá trình truy xuất, \textbf{hệ thống} cũng sẽ đảm bảo xử lý lỗi nếu phát sinh.
\end{itemize}

\subsection*{Mô tả sơ đồ hoạt động: Việc tải tài liệu xuống}
\begin{itemize}
    \item \textbf{Người dùng} chọn biểu tượng "Tải xuống" trên giao diện ứng dụng.
    \item Sau đó \textbf{hệ thống} kiểm tra quyền tải xuống rồi thực hiện truy xuất file cần tải xuống.
    \item Nếu có phát sinh lỗi, \textbf{hệ thống} sẽ hỏi xem \textbf{người dùng} muốn tiếp tục tải xuống lại nữa không, nếu người dùng muốn thì sẽ tiếp tục 
thử tải lại, còn nếu không thì về màn hình chính.
    \item Khi việc tải diễn ra thành công, file sẽ nằm trên máy \textbf{người dùng}.
\end{itemize}
